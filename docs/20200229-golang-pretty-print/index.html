一、我有一個 struct

stack overflow 上有重點！！ 

func prettyPrint(data interface{}) {
	jsonByte, err := json.MarshalIndent(data, "", "    ")
	if err != nil {
		fmt.Println("")
	}
	fmt.Printf("%s\n", jsonByte)
}

二、我有一個 json byte

func prettyPrintByte(jsonByte []byte) {
	var buf bytes.Buffer
	err := json.Indent(&buf, jsonByte, "", "    ")
	if err == nil {
		jsonByte = buf.Bytes()
	}
	fmt.Printf("%s\n", jsonByte)
}
